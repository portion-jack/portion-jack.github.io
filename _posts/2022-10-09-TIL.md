>소수 찾기

```
# -*- coding: utf-8 -*-
# UTF-8 encoding when using korean
def is_prime(num):
	if num <= 0 :
		return False
	if num == 1:
		return False
	if num ==2:
		return True
	for i in range(2,int(num**0.5)+1):
		if num % i == 0 :
			return False
	return True


user_input = input()
# print ("Hello Goorm! Your input is " + user_input)

sequence = list(map(int,input().split()))

ans = 0
for i in range(len(sequence)):
	if is_prime(i+1):
		ans += sequence[i]
		# print(sequence[i])
print(ans)

```
